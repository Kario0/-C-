2）ConfigManager.cpp
#include "ConfigManager.h"

#include <fstream>
#include <iostream>
#include <sstream>

bool ConfigManager::loadFromFile(const std::string& fileName)
{
    std::ifstream fin(fileName);

    if (!fin.is_open())
    {
        std::cout << "Failed to open config file: " << fileName << std::endl;
        return false;
    }

    params.clear();

    std::string line;
    while (std::getline(fin, line))
    {
        if (line.empty())
        {
            continue;
        }

        std::size_t pos = line.find('=');
        if (pos == std::string::npos)
        {
            continue;
        }

        std::string key = line.substr(0, pos);
        std::string valueText = line.substr(pos + 1);

        try
        {
            double value = std::stod(valueText);
            params[key] = value;
        }
        catch (...)
        {
            std::cout << "Invalid config line: " << line << std::endl;
        }
    }

    fin.close();
    return true;
}

bool ConfigManager::saveToFile(const std::string& fileName) const
{
    std::ofstream fout(fileName);

    if (!fout.is_open())
    {
        std::cout << "Failed to save config file: " << fileName << std::endl;
        return false;
    }

    for (const auto& item : params)
    {
        fout << item.first << "=" << item.second << std::endl;
    }

    fout.close();
    return true;
}

void ConfigManager::showAllParams() const
{
    if (params.empty())
    {
        std::cout << "No params found." << std::endl;
        return;
    }

    std::cout << "=========== CONFIG PARAMS ===========" << std::endl;
    for (const auto& item : params)
    {
        std::cout << item.first << " = " << item.second << std::endl;
    }
}

bool ConfigManager::hasParam(const std::string& key) const
{
    return params.find(key) != params.end();
}

double ConfigManager::getParam(const std::string& key) const
{
    auto it = params.find(key);
    if (it != params.end())
    {
        return it->second;
    }

    return 0.0;
}

bool ConfigManager::setParam(const std::string& key, double value)
{
    if (!isValidParam(key, value))
    {
        return false;
    }

    params[key] = value;
    return true;
}

bool ConfigManager::isValidParam(const std::string& key, double value) const
{
    if (key == "TempLimit")
    {
        return value >= 20.0 && value <= 100.0;
    }

    if (key == "PressureLimit")
    {
        return value >= 0.0 && value <= 5.0;
    }

    if (key == "OutputTarget")
    {
        return value >= 0.0 && value <= 100000.0;
    }

    if (key == "CommTimeout")
    {
        return value >= 1.0 && value <= 60.0;
    }

    if (key == "MaxRetry")
    {
        return value >= 1.0 && value <= 10.0;
    }

    return value >= 0.0;
}

4）MessageParser.cpp
#include "MessageParser.h"

std::vector<std::string> MessageParser::split(const std::string& text, char delim) const
{
    std::vector<std::string> parts;
    std::string current;

    for (char ch : text)
    {
        if (ch == delim)
        {
            parts.push_back(current);
            current.clear();
        }
        else
        {
            current += ch;
        }
    }

    parts.push_back(current);
    return parts;
}

std::map<std::string, std::string> MessageParser::parseMessage(const std::string& msg) const
{
    std::map<std::string, std::string> data;

    std::vector<std::string> fields = split(msg, ';');

    for (const auto& field : fields)
    {
        std::size_t pos = field.find('=');
        if (pos == std::string::npos)
        {
            continue;
        }

        std::string key = field.substr(0, pos);
        std::string value = field.substr(pos + 1);

        data[key] = value;
    }

    return data;
}

bool MessageParser::isValidMessage(const std::string& msg) const
{
    if (msg.empty())
    {
        return false;
    }

    if (msg.find('=') == std::string::npos)
    {
        return false;
    }

    if (msg.find("ID=") == std::string::npos)
    {
        return false;
    }

    return true;
}

#include <iostream>
#include <map>
#include <string>

#include "Device.h"
#include "DeviceManager.h"
#include "LogManager.h"
#include "ConfigManager.h"
#include "MessageParser.h"

int main()
{
    LogManager logMgr;
    DeviceManager devMgr;
    ConfigManager cfgMgr;
    MessageParser parser;

    logMgr.addInfo("System Start");

    if (cfgMgr.loadFromFile("config.txt"))
    {
        logMgr.addInfo("Config loaded successfully");
    }
    else
    {
        logMgr.addError("Config load failed");
    }

    cfgMgr.showAllParams();

    Device d1("01", "Bond_01");
    Device d2("02", "Bond_02");
    Device d3("03", "Bond_03");

    d1.setRunState("Idle");
    d2.setRunState("Idle");
    d3.setRunState("Idle");

    devMgr.addDevice(d1);
    devMgr.addDevice(d2);
    devMgr.addDevice(d3);

    logMgr.addInfo("Devices initialized");

    std::cout << "\n===== BEFORE MESSAGE UPDATE =====" << std::endl;
    devMgr.showAllDevices();

    std::string msg1 = "ID=01;TEMP=36.5;PRESS=1.10;ALARM=0";
    std::string msg2 = "ID=02;TEMP=41.2;PRESS=1.00;ALARM=1";
    std::string msg3 = "ID=03;TEMP=35.8;PRESS=1.50;ALARM=0";

    std::string messages[3] = { msg1, msg2, msg3 };

    for (const auto& msg : messages)
    {
        logMgr.addInfo("Receive message: " + msg);

        if (!parser.isValidMessage(msg))
        {
            logMgr.addError("Invalid message format: " + msg);
            continue;
        }

        std::map<std::string, std::string> data = parser.parseMessage(msg);

        if (data.find("ID") == data.end())
        {
            logMgr.addError("Message missing ID: " + msg);
            continue;
        }

        Device* dev = devMgr.findDeviceById(data["ID"]);
        if (dev == nullptr)
        {
            logMgr.addError("Device not found, ID=" + data["ID"]);
            continue;
        }

        try
        {
            if (data.find("TEMP") != data.end())
            {
                dev->setTemp(std::stod(data["TEMP"]));
            }

            if (data.find("PRESS") != data.end())
            {
                dev->setPressure(std::stod(data["PRESS"]));
            }

            dev->setOutput(dev->getOutput() + 10);

            double tempLimit = cfgMgr.getParam("TempLimit");
            double pressureLimit = cfgMgr.getParam("PressureLimit");

            dev->checkAlarm(tempLimit, pressureLimit);

            if (dev->isAlarm())
            {
                logMgr.addWarn(dev->getName() + " alarm: " + dev->getAlarmText());
            }
            else
            {
                logMgr.addInfo(dev->getName() + " update success, no alarm");
            }
        }
        catch (...)
        {
            logMgr.addError("Data convert failed in message: " + msg);
        }
    }

    std::cout << "\n===== AFTER MESSAGE UPDATE =====" << std::endl;
    devMgr.showAllDevices();

    std::cout << "\n===== SUMMARY =====" << std::endl;
    std::cout << "Device Count   : " << devMgr.getDeviceCount() << std::endl;
    std::cout << "Alarm Count    : " << devMgr.countAlarmDevices() << std::endl;
    std::cout << "Average Output : " << devMgr.getAverageOutput() << std::endl;

    Device* maxTempDev = devMgr.getMaxTempDevice();
    if (maxTempDev != nullptr)
    {
        std::cout << "Max Temp Device: " << maxTempDev->getName()
                  << " (" << maxTempDev->getTemp() << ")" << std::endl;
    }

    std::cout << "\n===== LOGS =====" << std::endl;
    logMgr.showAllLogs();

    logMgr.saveToFile("log.txt");
    logMgr.addInfo("System Finished");

    return 0;
}
