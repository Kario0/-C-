4）CommManager.cpp
#include "CommManager.h"

#include <iostream>

CommManager::CommManager(LogManager* logMgr, DeviceManager* devMgr)
    : logMgr(logMgr),
      devMgr(devMgr)
{
}

void CommManager::sendCommand(Command& cmd)
{
    if (logMgr != nullptr)
    {
        logMgr->addInfo("Send Command -> TargetID=" + cmd.getTargetId() +
                        ", CMD=" + cmd.getCmdName());
    }
}

bool CommManager::matchResponse(const Command& cmd,
                                const std::map<std::string, std::string>& data) const
{
    auto itId = data.find("ID");
    auto itCmd = data.find("CMD");
    auto itResult = data.find("RESULT");

    if (itId == data.end() || itCmd == data.end() || itResult == data.end())
    {
        return false;
    }

    if (itId->second != cmd.getTargetId())
    {
        return false;
    }

    if (itCmd->second != cmd.getCmdName())
    {
        return false;
    }

    if (itResult->second != "OK")
    {
        return false;
    }

    return true;
}

std::string CommManager::getNextState(const std::string& currentState,
                                      const std::string& cmdName,
                                      bool success) const
{
    if (!success)
    {
        return "Alarm";
    }

    if (cmdName == "START")
    {
        return "Running";
    }

    if (cmdName == "STOP")
    {
        return "Stop";
    }

    if (cmdName == "RESET")
    {
        return "Idle";
    }

    return currentState;
}

bool CommManager::processResponse(const std::string& response, Command& cmd)
{
    if (!parser.isValidMessage(response))
    {
        if (logMgr != nullptr)
        {
            logMgr->addError("Invalid response format: " + response);
        }
        return false;
    }

    std::map<std::string, std::string> data = parser.parseMessage(response);

    if (!matchResponse(cmd, data))
    {
        if (logMgr != nullptr)
        {
            logMgr->addWarn("Response not matched for CMD=" + cmd.getCmdName() +
                            ", Response=" + response);
        }
        return false;
    }

    Device* dev = nullptr;
    if (devMgr != nullptr)
    {
        dev = devMgr->findDeviceById(cmd.getTargetId());
    }

    if (dev == nullptr)
    {
        if (logMgr != nullptr)
        {
            logMgr->addError("Device not found when processing response, ID=" + cmd.getTargetId());
        }
        return false;
    }

    cmd.setDone(true);

    std::string nextState = getNextState(dev->getRunState(), cmd.getCmdName(), true);
    dev->setRunState(nextState);

    if (cmd.getCmdName() == "RESET")
    {
        dev->clearAlarm();
    }

    if (logMgr != nullptr)
    {
        logMgr->addInfo("Response matched success -> Device=" + dev->getName() +
                        ", CMD=" + cmd.getCmdName() +
                        ", NewState=" + dev->getRunState());
    }

    return true;
}

void CommManager::handleTimeout(Command& cmd)
{
    Device* dev = nullptr;
    if (devMgr != nullptr)
    {
        dev = devMgr->findDeviceById(cmd.getTargetId());
    }

    if (dev != nullptr)
    {
        dev->setRunState("Alarm");
    }

    if (logMgr != nullptr)
    {
        logMgr->addError("Command timeout -> TargetID=" + cmd.getTargetId() +
                         ", CMD=" + cmd.getCmdName());
    }
}

bool CommManager::waitForResponse(Command& cmd,
                                  const std::string& simulatedResponse,
                                  int maxRetry)
{
    while (cmd.getRetryCount() < maxRetry)
    {
        if (!simulatedResponse.empty())
        {
            bool ok = processResponse(simulatedResponse, cmd);
            if (ok)
            {
                return true;
            }
        }

        cmd.increaseRetry();

        if (logMgr != nullptr)
        {
            logMgr->addWarn("Retry command -> TargetID=" + cmd.getTargetId() +
                            ", CMD=" + cmd.getCmdName() +
                            ", Retry=" + std::to_string(cmd.getRetryCount()));
        }
    }

    handleTimeout(cmd);
    return false;
}
5）接入通信后的 main.cpp

这版是在你上一版基础上，继续把 Command 和 CommManager 接进来。

#include <iostream>
#include <map>
#include <string>

#include "Device.h"
#include "DeviceManager.h"
#include "LogManager.h"
#include "ConfigManager.h"
#include "MessageParser.h"
#include "Command.h"
#include "CommManager.h"

int main()
{
    LogManager logMgr;
    DeviceManager devMgr;
    ConfigManager cfgMgr;
    MessageParser parser;

    logMgr.addInfo("System Start");

    if (cfgMgr.loadFromFile("config.txt"))
    {
        logMgr.addInfo("Config loaded successfully");
    }
    else
    {
        logMgr.addError("Config load failed");
    }

    Device d1("01", "Bond_01");
    Device d2("02", "Bond_02");
    Device d3("03", "Bond_03");

    d1.setRunState("Idle");
    d2.setRunState("Idle");
    d3.setRunState("Idle");

    devMgr.addDevice(d1);
    devMgr.addDevice(d2);
    devMgr.addDevice(d3);

    logMgr.addInfo("Devices initialized");

    std::cout << "\n===== BEFORE MESSAGE UPDATE =====" << std::endl;
    devMgr.showAllDevices();

    std::string msg1 = "ID=01;TEMP=36.5;PRESS=1.10;ALARM=0";
    std::string msg2 = "ID=02;TEMP=41.2;PRESS=1.00;ALARM=1";
    std::string msg3 = "ID=03;TEMP=35.8;PRESS=1.50;ALARM=0";

    std::string messages[3] = {msg1, msg2, msg3};

    for (const auto& msg : messages)
    {
        logMgr.addInfo("Receive message: " + msg);

        if (!parser.isValidMessage(msg))
        {
            logMgr.addError("Invalid message format: " + msg);
            continue;
        }

        std::map<std::string, std::string> data = parser.parseMessage(msg);

        if (data.find("ID") == data.end())
        {
            logMgr.addError("Message missing ID: " + msg);
            continue;
        }

        Device* dev = devMgr.findDeviceById(data["ID"]);
        if (dev == nullptr)
        {
            logMgr.addError("Device not found, ID=" + data["ID"]);
            continue;
        }

        try
        {
            if (data.find("TEMP") != data.end())
            {
                dev->setTemp(std::stod(data["TEMP"]));
            }

            if (data.find("PRESS") != data.end())
            {
                dev->setPressure(std::stod(data["PRESS"]));
            }

            dev->setOutput(dev->getOutput() + 10);

            double tempLimit = cfgMgr.getParam("TempLimit");
            double pressureLimit = cfgMgr.getParam("PressureLimit");

            dev->checkAlarm(tempLimit, pressureLimit);

            if (dev->isAlarm())
            {
                logMgr.addWarn(dev->getName() + " alarm: " + dev->getAlarmText());
            }
            else
            {
                logMgr.addInfo(dev->getName() + " update success, no alarm");
            }
        }
        catch (...)
        {
            logMgr.addError("Data convert failed in message: " + msg);
        }
    }

    std::cout << "\n===== AFTER MESSAGE UPDATE =====" << std::endl;
    devMgr.showAllDevices();

    CommManager commMgr(&logMgr, &devMgr);

    std::cout << "\n===== COMMAND TEST =====" << std::endl;

    // 成功案例
    Command startCmd("01", "START");
    commMgr.sendCommand(startCmd);

    std::string okResponse = "ID=01;CMD=START;RESULT=OK";
    bool startOk = commMgr.waitForResponse(startCmd, okResponse,
                                           static_cast<int>(cfgMgr.getParam("MaxRetry")));

    std::cout << "START Command Result: " << (startOk ? "SUCCESS" : "FAIL") << std::endl;

    // 失败案例：故意给一个不匹配响应，触发重试和超时
    Command stopCmd("02", "STOP");
    commMgr.sendCommand(stopCmd);

    std::string badResponse = "ID=02;CMD=RESET;RESULT=FAIL";
    bool stopOk = commMgr.waitForResponse(stopCmd, badResponse,
                                          static_cast<int>(cfgMgr.getParam("MaxRetry")));

    std::cout << "STOP Command Result : " << (stopOk ? "SUCCESS" : "FAIL") << std::endl;

    std::cout << "\n===== FINAL DEVICE STATE =====" << std::endl;
    devMgr.showAllDevices();

    std::cout << "\n===== SUMMARY =====" << std::endl;
    std::cout << "Device Count   : " << devMgr.getDeviceCount() << std::endl;
    std::cout << "Alarm Count    : " << devMgr.countAlarmDevices() << std::endl;
    std::cout << "Average Output : " << devMgr.getAverageOutput() << std::endl;

    Device* maxTempDev = devMgr.getMaxTempDevice();
    if (maxTempDev != nullptr)
    {
        std::cout << "Max Temp Device: " << maxTempDev->getName()
                  << " (" << maxTempDev->getTemp() << ")" << std::endl;
    }

    std::cout << "\n===== LOGS =====" << std::endl;
    logMgr.showAllLogs();

    logMgr.addInfo("System Finished");
    logMgr.saveToFile("log.txt");

    return 0;
}
