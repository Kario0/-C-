void Device::checkAlarm(double tempLimit, double pressureLimit)
{
    if (temp > tempLimit)
    {
        alarmState = true;
        alarmCode = 1001;
        alarmText = "Temperature Over Limit";
        runState = "Alarm";
    }
    else if (pressure > pressureLimit)
    {
        alarmState = true;
        alarmCode = 1002;
        alarmText = "Pressure Over Limit";
        runState = "Alarm";
    }
    else
    {
        clearAlarm();
    }
}

void Device::clearAlarm()
{
    alarmState = false;
    alarmCode = 0;
    alarmText = "No Alarm";

    if (runState == "Alarm")
    {
        runState = "Idle";
    }
}

void Device::showInfo() const
{
    std::cout << "----------------------------------------" << std::endl;
    std::cout << "Device ID   : " << id << std::endl;
    std::cout << "Device Name : " << name << std::endl;
    std::cout << "Temp        : " << std::fixed << std::setprecision(2) << temp << std::endl;
    std::cout << "Pressure    : " << std::fixed << std::setprecision(2) << pressure << std::endl;
    std::cout << "Output      : " << output << std::endl;
    std::cout << "Run State   : " << runState << std::endl;
    std::cout << "Alarm State : " << (alarmState ? "ON" : "OFF") << std::endl;
    std::cout << "Alarm Code  : " << alarmCode << std::endl;
    std::cout << "Alarm Text  : " << alarmText << std::endl;
}

#include "DeviceManager.h"

#include <iostream>

void DeviceManager::addDevice(const Device& device)
{
    devices.push_back(device);
}

Device* DeviceManager::findDeviceById(const std::string& id)
{
    for (auto& device : devices)
    {
        if (device.getId() == id)
        {
            return &device;
        }
    }

    return nullptr;
}

Device* DeviceManager::findDeviceByName(const std::string& name)
{
    for (auto& device : devices)
    {
        if (device.getName() == name)
        {
            return &device;
        }
    }

    return nullptr;
}

void DeviceManager::showAllDevices() const
{
    if (devices.empty())
    {
        std::cout << "No devices found." << std::endl;
        return;
    }

    for (const auto& device : devices)
    {
        device.showInfo();
    }
}

int DeviceManager::countAlarmDevices() const
{
    int count = 0;

    for (const auto& device : devices)
    {
        if (device.isAlarm())
        {
            count++;
        }
    }

    return count;
}

double DeviceManager::getAverageOutput() const
{
    if (devices.empty())
    {
        return 0.0;
    }

    int total = 0;

    for (const auto& device : devices)
    {
        total += device.getOutput();
    }

    return static_cast<double>(total) / devices.size();
}

Device* DeviceManager::getMaxTempDevice()
{
    if (devices.empty())
    {
        return nullptr;
    }

    Device* maxDevice = &devices[0];

    for (auto& device : devices)
    {
        if (device.getTemp() > maxDevice->getTemp())
        {
            maxDevice = &device;
        }
    }

    return maxDevice;
}

std::size_t DeviceManager::getDeviceCount() const
{
    return devices.size();
}

void LogManager::addLog(const std::string& level, const std::string& msg)
{
    std::string line = "[" + level + "] " + msg;
    logs.push_back(line);
}

void LogManager::addInfo(const std::string& msg)
{
    addLog("INFO", msg);
}

void LogManager::addWarn(const std::string& msg)
{
    addLog("WARN", msg);
}

void LogManager::addError(const std::string& msg)
{
    addLog("ERROR", msg);
}

void LogManager::showAllLogs() const
{
    if (logs.empty())
    {
        std::cout << "No logs found." << std::endl;
        return;
    }

    std::cout << "=========== ALL LOGS ===========" << std::endl;

    for (const auto& log : logs)
    {
        std::cout << log << std::endl;
    }
}

void LogManager::showLogsByKeyword(const std::string& keyword) const
{
    std::cout << "======= LOGS BY KEYWORD: " << keyword << " =======" << std::endl;

    bool found = false;

    for (const auto& log : logs)
    {
        if (log.find(keyword) != std::string::npos)
        {
            std::cout << log << std::endl;
            found = true;
        }
    }

    if (!found)
    {
        std::cout << "No matched logs." << std::endl;
    }
}

void LogManager::showLogsByLevel(const std::string& level) const
{
    std::string tag = "[" + level + "]";

    std::cout << "======= LOGS BY LEVEL: " << level << " =======" << std::endl;

    bool found = false;

    for (const auto& log : logs)
    {
        if (log.find(tag) != std::string::npos)
        {
            std::cout << log << std::endl;
            found = true;
        }
    }

    if (!found)
    {
        std::cout << "No matched logs." << std::endl;
    }
}

void LogManager::saveToFile(const std::string& fileName) const
{
    std::ofstream fout(fileName, std::ios::app);

    if (!fout.is_open())
    {
        std::cout << "Failed to open log file: " << fileName << std::endl;
        return;
    }

    for (const auto& log : logs)
    {
        fout << log << std::endl;
    }

    fout.close();
}

std::size_t LogManager::getLogCount() const
{
    return logs.size();
}
7）main.cpp（测试主程序）

这份 main.cpp 是为了让你先把第一版跑起来。

#include <iostream>

#include "Device.h"
#include "DeviceManager.h"
#include "LogManager.h"

int main()
{
    LogManager logMgr;
    DeviceManager devMgr;

    logMgr.addInfo("System Start");

    Device d1("01", "Bond_01");
    Device d2("02", "Bond_02");
    Device d3("03", "Bond_03");

    d1.setTemp(36.5);
    d1.setPressure(1.10);
    d1.setOutput(120);
    d1.setRunState("Idle");

    d2.setTemp(41.2);
    d2.setPressure(1.00);
    d2.setOutput(100);
    d2.setRunState("Running");

    d3.setTemp(35.8);
    d3.setPressure(1.50);
    d3.setOutput(130);
    d3.setRunState("Running");

    devMgr.addDevice(d1);
    devMgr.addDevice(d2);
    devMgr.addDevice(d3);

    logMgr.addInfo("Devices initialized");

    const double tempLimit = 38.0;
    const double pressureLimit = 1.20;

    Device* p1 = devMgr.findDeviceById("01");
    Device* p2 = devMgr.findDeviceById("02");
    Device* p3 = devMgr.findDeviceById("03");

    if (p1 != nullptr)
    {
        p1->checkAlarm(tempLimit, pressureLimit);
        if (p1->isAlarm())
        {
            logMgr.addWarn(p1->getName() + " alarm: " + p1->getAlarmText());
        }
    }

    if (p2 != nullptr)
    {
        p2->checkAlarm(tempLimit, pressureLimit);
        if (p2->isAlarm())
        {
            logMgr.addWarn(p2->getName() + " alarm: " + p2->getAlarmText());
        }
    }

    if (p3 != nullptr)
    {
        p3->checkAlarm(tempLimit, pressureLimit);
        if (p3->isAlarm())
        {
            logMgr.addWarn(p3->getName() + " alarm: " + p3->getAlarmText());
        }
    }

    std::cout << "\n===== DEVICE LIST =====" << std::endl;
    devMgr.showAllDevices();

    std::cout << "\n===== SUMMARY =====" << std::endl;
    std::cout << "Device Count   : " << devMgr.getDeviceCount() << std::endl;
    std::cout << "Alarm Count    : " << devMgr.countAlarmDevices() << std::endl;
    std::cout << "Average Output : " << devMgr.getAverageOutput() << std::endl;

    Device* maxTempDev = devMgr.getMaxTempDevice();
    if (maxTempDev != nullptr)
    {
        std::cout << "Max Temp Device: " << maxTempDev->getName()
                  << " (" << maxTempDev->getTemp() << ")" << std::endl;
    }

    logMgr.addInfo("System Finished");
    logMgr.saveToFile("log.txt");

    std::cout << "\n===== LOGS =====" << std::endl;
    logMgr.showAllLogs();

    return 0;
}

10）你写这一版时，重点检查这 5 个点
1. findDeviceById() 返回的是不是设备指针

因为后面你要用它直接更新设备。

2. checkAlarm() 是不是只负责报警判断

不要把太多别的逻辑塞进去。

3. main() 里是不是只做流程串联

不要让 main() 变成超大函数。

4. LogManager 是否统一接管日志

不要到处乱 cout 当日志。

5. 变量命名是否清楚

例如：

tempLimit
pressureLimit
alarmState
runState
